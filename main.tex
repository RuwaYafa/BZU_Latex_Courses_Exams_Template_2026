% 2. Answer Space Control
%    Increase or decrease this value to scale all blank answer spaces.
% =========================================================================
\newcommand{\AnswerSpaceScale}{7}

% =========================================================================
% 3. Mode Flags
%    These flags control which version is printed:
%    - Question Sheet
%    - Blank Answer Sheet
%    - Answer Key
% =========================================================================
\newboolean{isAnswerMode}
\newboolean{isKeyMode}

\setboolean{isAnswerMode}{false}
\setboolean{isKeyMode}{false}

% Shows question text only in the Question Sheet.
% Hides question text in the Answer Sheet and Answer Key.
\newcommand{\questiontext}[1]{%
  \ifthenelse{\boolean{isAnswerMode} \OR \boolean{isKeyMode}}{}{#1}%
}

% Adds blank answer space only in the Answer Sheet.
\newcommand{\answerspace}[1]{%
  \ifthenelse{\boolean{isAnswerMode}}{\vspace{#1}}{}%
}

% =========================================================================
% 4. Question Environment
% =========================================================================
\newcounter{examqmarks}

\NewDocumentEnvironment{examquestion}{m m O{}}{%
  % #1 = question number
  % #2 = marks
  % #3 = optional fixed answer space override, e.g. 4cm
  \setcounter{examqmarks}{#2}%
  \noindent\textbf{Question #1 \normalfont(#2 Marks):}%
  \ifthenelse{\boolean{isAnswerMode} \OR \boolean{isKeyMode}}{%
    \par\vspace{0.1cm}%
  }{%
    \space
  }%
}{%
  \ifthenelse{\boolean{isAnswerMode}}{%
    \ifthenelse{\equal{#3}{}}{%
      \answerspace{\the\numexpr\value{examqmarks}*\AnswerSpaceScale\relax pt}%
    }{%
      \answerspace{#3}%
    }%
  }{%
    \vspace{0.1cm}%
  }%
}

% =========================================================================
% 5. Subpart Environment
% =========================================================================
\newlist{subparts}{enumerate}{1}
\setlist[subparts]{
  label=(\alph*),
  leftmargin=1.8em,
  topsep=0pt,
  itemsep=6pt,
  parsep=0pt
}

\NewDocumentCommand{\subpart}{m m O{}}{%
  % #1 = subpart text
  % #2 = marks
  % #3 = optional fixed answer space override
  \item
  \ifthenelse{\boolean{isAnswerMode} \OR \boolean{isKeyMode}}{%
    \ifthenelse{\boolean{isAnswerMode}}{%
      \ifthenelse{\equal{#3}{}}{%
        \answerspace{\the\numexpr#2*\AnswerSpaceScale\relax pt}%
      }{%
        \answerspace{#3}%
      }%
      \vspace{0.2cm}%
    }{}%
  }{%
    #1 \textit{(#2 marks)}%
    \vspace{0.15cm}%
  }%
}
